import re, os, unittest


def create_short_bios(lines_list):
    bio_list = []
    expression1 = 'Bio: (.+[\\.!|?])'
    for line in lines_list:
        line.rstrip()
        bio_first = re.findall(expression1, line)
        bio_list.extend(bio_first)
    return bio_list
import re, os, unittest


def create_short_bios(lines_list):
    bio_list = []
    expression1 = 'Bio: (.+?[\\.!\\?])'
    for line in lines_list:
        line.rstrip()
        bio_first = re.findall(expression1, line)
        if bio_first:
            for m in bio_first:
                bio_list.append(m)
    return bio_list
