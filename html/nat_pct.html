def nat_pct(data, col_list):
    data_totals = {}
    for demographic, regions in data.items():
        if demographic in col_list:
            for region, count in region.items():
                if region == 'Region Totals':
                    continue
                if demographic not in data_totals:
                    data_totals[demographic] = 0
                data_totals[demographic] += count
    total_count = sum(data_totals.values())
    for demographic, count in data_totals.items():
        data_totals[demographic] = round(count / total_count * 100, 2)
    return data_totals
