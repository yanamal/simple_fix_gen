import io
import sys
import csv
import unittest


def load_csv(filename):
    inFile = open(filename)
    csvFile = csv.reader(inFile)
    headers = next(csvFile)
    data = {}
    ethnicity_dict = {}
    for header in headers[1:]:
        ethnicity_dict[header] = {}
    for row in csvFile:
        region = row[0]
        data[region] = {}
        for i in range(1, len(row) + 1):
            data_val = row[i]
            ethnicity_dict[headers[i]] = data_val
            data[region] = ethnicity_dict
    inFile.close()
    print(data)
    return data
