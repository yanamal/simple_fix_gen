def min_max(data):
    min_max = {'min': {}, 'max': {}}
    for dem in data:
        min_max['min'][dem] = {}
        min_max['max'][dem] = {}
        dem_sorted = sorted(data[dem], key=data[dem].get, reverse=False)
        min_max['min'][dem][dem_sorted[0]] = data[dem][dem_sorted[0]]
        min_max['max'][dem][dem_sorted[-1]] = data[dem][dem_sorted[-1]]
    return min_max
