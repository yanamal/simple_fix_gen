def calculate_in_middle(mid, lines_list):
    expression = '\\b[A-Za-z]+' + re.escape(mid) + '[A-Za-z]+\\b'
    mid_list = []
    for line in lines_list:
        mid_word = re.findall(expression, line)
        for i in mid_word:
            mid_list.extend(i[0])
    count = len(mid_list)
    return count
